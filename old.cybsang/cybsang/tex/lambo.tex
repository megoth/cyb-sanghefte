\spvers{
\alle{	Se der sitter en fyllehund, \\
	Mine herrer Lambo. \\
	Sett nu glasset for din mund, \\
	Mine herrer Lambo. \\
	Se hvordan hver dråpe vanker \\
	Nedad halsen på den dranker. \\
	Lambo, Lambo, \\
	Mine herrer Lambo.}
\vspace*{6mm}}

\spvers{ %\vspace*{5mm}
\ene{Jeg mitt glass utdrukket har,}
\alle{Mine herrer Lambo.}
\ene{Se der fins ei dråpe kvar,}
\alle{mine herrer Lambo.}
\ene{	Som bevis derpå jeg vende \\
	Glasset på dets annen ende.}
\alle{ 	Lambo, Lambo, \\
	Mine herrer Lambo.}
\vspace*{6mm}}
